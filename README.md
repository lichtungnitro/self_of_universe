class Nature:
	'''
	Define Nature
	'''
	def __init__(self):
		self.each = True
		self.other = False

	def denial(self):
		self.each = not self.each
		self.other = not self.other
		return {
			"each": self.each
			"other": self.other,
			}

	def switch(self):
		self.other, self.each = self.each, self.other
		return {
			"other": self.other,
			"each": self.each
			} 

	def maintain(self):
		return {
			"each": self.each,
			"other": self.other
			}

if __name__ == "__main__":
	# Initialize Entities
	Each = Nature()
	Other = Nature()

	# Initialize Entities Again
	Other = Nature()
	Each = Nature()

	# Proof of Rule Switch
	rule_of_self_switch = \
	Other.switch()["each"] == Each.maintain()["other"]
